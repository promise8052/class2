#!/usr/bin/env python3 """ 기체의 성질 마무리 퀴즈 - GitHub Pages 자동 배포 스크립트 사용법: python3 GitHub_배포스크립트.py """ import urllib.request import urllib.error import json import base64 import os import sys # ───────────────────────────────────────── # ★ 여기만 수정하세요 ★ GITHUB_USERNAME = "" # 예: teacher123 GITHUB_TOKEN = "" # GitHub Personal Access Token REPO_NAME = "quiz" # 저장소 이름 (영문) HTML_FILE = "기체의성질_마무리퀴즈.html" # 같은 폴더에 있어야 함 # ───────────────────────────────────────── def api_call(method, endpoint, data=None, token=None): url = f"https://api.github.com{endpoint}" headers = { "Accept": "application/vnd.github+json", "X-GitHub-Api-Version": "2022-11-28", } if token: headers["Authorization"] = f"Bearer {token}" if data: headers["Content-Type"] = "application/json" body = json.dumps(data).encode() if data else None req = urllib.request.Request(url, data=body, headers=headers, method=method) try: with urllib.request.urlopen(req) as res: return json.loads(res.read()), res.status except urllib.error.HTTPError as e: err_body = e.read().decode() return json.loads(err_body) if err_body else {}, e.code def deploy(): print("\n" + "="*50) print(" 기체의 성질 퀴즈 — GitHub Pages 자동 배포") print("="*50) # 입력값 확인 username = GITHUB_USERNAME.strip() token = GITHUB_TOKEN.strip() repo = REPO_NAME.strip() if not username or not token: print("\n❌ GITHUB_USERNAME과 GITHUB_TOKEN을 스크립트 상단에 입력해주세요.") print("\n[ GitHub Token 발급 방법 ]") print("1. github.com 로그인") print("2. 오른쪽 위 프로필 → Settings") print("3. 왼쪽 메뉴 맨 아래 → Developer settings") print("4. Personal access tokens → Tokens (classic)") print("5. Generate new token (classic)") print("6. Note: quiz-deploy / Expiration: 90 days") print("7. Scope: ✅ repo 전체 체크") print("8. Generate token → 토큰 복사해서 위에 붙여넣기") input("\n[Enter]키를 눌러 종료...") return # HTML 파일 읽기 script_dir = os.path.dirname(os.path.abspath(__file__)) html_path = os.path.join(script_dir, HTML_FILE) if not os.path.exists(html_path): print(f"\n❌ 파일을 찾을 수 없어요: {html_path}") print(f" 이 스크립트와 같은 폴더에 '{HTML_FILE}' 파일이 있어야 해요.") input("\n[Enter]키를 눌러 종료...") return print(f"\n✅ 파일 확인: {HTML_FILE}") with open(html_path, 'rb') as f: html_b64 = base64.b64encode(f.read()).decode() # 1) 저장소 존재 확인 print(f"\n🔍 저장소 확인 중... ({username}/{repo})") res, status = api_call("GET", f"/repos/{username}/{repo}", token=token) if status == 404: # 저장소 생성 print(f"📁 저장소 생성 중...") res, status = api_call("POST", "/user/repos", { "name": repo, "description": "기체의 성질 마무리 퀴즈", "private": False, "auto_init": False, }, token=token) if status not in (200, 201): print(f"❌ 저장소 생성 실패: {res.get('message','')}") input("\n[Enter]키를 눌러 종료...") return print(f"✅ 저장소 생성 완료!") import time; time.sleep(2) elif status == 200: print(f"✅ 기존 저장소 발견!") else: print(f"❌ 저장소 확인 실패 (status {status}): {res.get('message','')}") input("\n[Enter]키를 눌러 종료...") return # 2) 기존 index.html SHA 확인 (업데이트 시 필요) sha = None res2, status2 = api_call("GET", f"/repos/{username}/{repo}/contents/index.html", token=token) if status2 == 200: sha = res2.get("sha") print(f"🔄 기존 파일 업데이트 모드") else: print(f"📄 새 파일 업로드 모드") # 3) index.html 업로드 print(f"\n📤 퀴즈 HTML 업로드 중...") payload = { "message": "퀴즈 업데이트", "content": html_b64, "branch": "main", } if sha: payload["sha"] = sha res3, status3 = api_call("PUT", f"/repos/{username}/{repo}/contents/index.html", payload, token=token) if status3 not in (200, 201): # main 브랜치 없으면 master로 재시도 payload["branch"] = "master" res3, status3 = api_call("PUT", f"/repos/{username}/{repo}/contents/index.html", payload, token=token) if status3 not in (200, 201): print(f"❌ 파일 업로드 실패: {res3.get('message','')}") input("\n[Enter]키를 눌러 종료...") return print(f"✅ 파일 업로드 완료!") # 4) GitHub Pages 활성화 print(f"\n🌐 GitHub Pages 활성화 중...") branch = payload["branch"] res4, status4 = api_call("POST", f"/repos/{username}/{repo}/pages", { "source": {"branch": branch, "path": "/"} }, token=token) if status4 in (200, 201): print(f"✅ GitHub Pages 활성화 완료!") elif status4 == 409: print(f"✅ GitHub Pages 이미 활성화되어 있어요!") elif status4 == 422: # 이미 있는 경우 PUT으로 업데이트 api_call("PUT", f"/repos/{username}/{repo}/pages", { "source": {"branch": branch, "path": "/"} }, token=token) print(f"✅ GitHub Pages 설정 업데이트!") else: print(f"⚠️ Pages 설정 필요 (status {status4})") print(f" → github.com/{username}/{repo} → Settings → Pages → Source: {branch} 선택") # 5) 최종 URL page_url = f"https://{username}.github.io/{repo}" print(f"\n" + "="*50) print(f"🎉 배포 완료!") print(f" URL: {page_url}") print(f" ※ 처음 배포 시 1~3분 후 접속 가능해요") print("="*50) # 6) QR 코드 생성 try: import qrcode as qc from PIL import Image, ImageDraw, ImageFont import io print(f"\n🎨 QR 코드 생성 중...") qr = qc.QRCode(version=None, error_correction=qc.constants.ERROR_CORRECT_H, box_size=12, border=4) qr.add_data(page_url) qr.make(fit=True) qr_img = qr.make_image(fill_color="#1A237E", back_color="white").convert("RGB") # 캔버스에 QR + 텍스트 합성 qr_size = qr_img.size[0] canvas_w = qr_size + 80 canvas_h = qr_size + 180 canvas = Image.new("RGB", (canvas_w, canvas_h), "#F0F8FF") # QR 붙이기 qr_x = (canvas_w - qr_size) // 2 canvas.paste(qr_img, (qr_x, 40)) # URL 텍스트 draw = ImageDraw.Draw(canvas) try: font_big = ImageFont.truetype("/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf", 18) font_small= ImageFont.truetype("/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf", 14) except: font_big = ImageFont.load_default() font_small= font_big title = "기체의 성질 마무리 퀴즈" url_text = page_url # 제목 try: tw = draw.textlength(title, font=font_big) except: tw = len(title) * 11 draw.text(((canvas_w - tw) // 2, qr_size + 55), title, fill="#1A237E", font=font_big) # URL try: uw = draw.textlength(url_text, font=font_small) except: uw = len(url_text) * 8 draw.text(((canvas_w - uw) // 2, qr_size + 85), url_text, fill="#0288D1", font=font_small) # 안내 guide = "QR 스캔 후 퀴즈 시작!" try: gw = draw.textlength(guide, font=font_small) except: gw = len(guide) * 8 draw.text(((canvas_w - gw) // 2, qr_size + 112), guide, fill="#455A64", font=font_small) # 저장 qr_path = os.path.join(script_dir, "퀴즈_QR코드.png") canvas.save(qr_path, format="PNG", dpi=(300, 300)) print(f"✅ QR 코드 저장: 퀴즈_QR코드.png") print(f"\n📋 이 QR 이미지를 출력하거나 화면에 띄워서 학생들에게 보여주세요!") # 미리보기 열기 시도 try: if sys.platform == "darwin": os.system(f"open '{qr_path}'") elif sys.platform == "win32": os.startfile(qr_path) else: os.system(f"xdg-open '{qr_path}' 2>/dev/null &") except: pass except ImportError: print(f"\n⚠️ QR 이미지 자동 생성 실패 (qrcode 라이브러리 없음)") print(f" 아래 URL을 직접 QR 생성 사이트에 넣어주세요:") print(f" → {page_url}") print(f"\n 무료 QR 생성 사이트: https://qr.io 또는 https://qrcode-monkey.com") print(f"\n" + "="*50) input("완료! [Enter]키를 눌러 종료...") if __name__ == "__main__": deploy()
